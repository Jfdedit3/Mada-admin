local Runtime = {
	Connections = {},
	Loops = {},
	CleanupCallbacks = {},
	State = {},
	Flags = {}
}

local function SafeDisconnect(Connection)
	if Connection then
		pcall(function()
			Connection:Disconnect()
		end)
	end
end

function Runtime.BindConnection(Name, Signal, Callback)
	if Runtime.Connections[Name] then
		SafeDisconnect(Runtime.Connections[Name])
	end

	local Connection = Signal:Connect(Callback)
	Runtime.Connections[Name] = Connection
	return Connection
end

function Runtime.TrackConnection(Name, Connection)
	if Runtime.Connections[Name] then
		SafeDisconnect(Runtime.Connections[Name])
	end

	Runtime.Connections[Name] = Connection
	return Connection
end

function Runtime.Disconnect(Name)
	if Runtime.Connections[Name] then
		SafeDisconnect(Runtime.Connections[Name])
		Runtime.Connections[Name] = nil
	end
end

function Runtime.StartLoop(Name, Callback)
	if Runtime.Loops[Name] then
		Runtime.Loops[Name].Active = false
	end

	local Token = {Active = true}
	Runtime.Loops[Name] = Token

	task.spawn(function()
		local success, err =
			pcall(function()
				Callback(Token)
			end)

		if not success then
			warn("[Mada Admin] Loop '" .. tostring(Name) .. "' failed:", err)
		end

		if Runtime.Loops[Name] == Token then
			Runtime.Loops[Name] = nil
		end
	end)

	return Token
end

function Runtime.StopLoop(Name)
	if Runtime.Loops[Name] then
		Runtime.Loops[Name].Active = false
		Runtime.Loops[Name] = nil
	end
end

function Runtime.RegisterCleanup(Name, Callback)
	Runtime.CleanupCallbacks[Name] = Callback
	return Callback
end

function Runtime.RunCleanup(Name)
	local Callback = Runtime.CleanupCallbacks[Name]
	if not Callback then
		return
	end

	Runtime.CleanupCallbacks[Name] = nil

	local success, err = pcall(Callback)
	if not success then
		warn("[Mada Admin] Cleanup '" .. tostring(Name) .. "' failed:", err)
	end
end

function Runtime.CleanupAll()
	for Name in next, Runtime.Loops do
		Runtime.StopLoop(Name)
	end

	for Name in next, Runtime.Connections do
		Runtime.Disconnect(Name)
	end

	for Name in next, Runtime.CleanupCallbacks do
		Runtime.RunCleanup(Name)
	end
end

local function SafeGetObjects(AssetId)
	local success, Objects =
		pcall(function()
			return game:GetObjects("rbxassetid://" .. tostring(AssetId))
		end)

	if success and type(Objects) == "table" and Objects[1] then
		return Objects
	end

	error("Failed to load asset rbxassetid://" .. tostring(AssetId))
end

function UIFuncs.GetNotificationTemplate()
	local Handler = UIFuncs.NotificationHandler
	if not Handler then
		return nil
	end

	local DirectTemplate = Handler:FindFirstChild("Notification")

	end

	end

function UIFuncs.LoadSavedSettings()
	if not Checks.CheckIfFileFuncs then
		return
	end

	local SettingsPath = "Mada-Admin-V2/settings.json"
	if not isfile(SettingsPath) then
		return
	end

	local success, Decoded =
		pcall(function()
			return Services.HttpService:JSONDecode(readfile(SettingsPath))
		end)

	if success and type(Decoded) == "table" then
		if type(Decoded.Prefix) == "string" and Decoded.Prefix ~= "" then
			Settings.Prefix = Decoded.Prefix
		end
		if type(Decoded.GetPlayerSeperator) == "string" and Decoded.GetPlayerSeperator ~= "" then
			Settings.GetPlayerSeperator = Decoded.GetPlayerSeperator
		end
		if type(Decoded.Options) == "table" then
			for Key, Value in next, Decoded.Options do
				Settings.Options[Key] = Value
			end
		end
	end

	UIFuncs.ApplySettings()
end

function UIFuncs.SaveSettings()
	if not Checks.CheckIfFileFuncs then
		return false
	end

	local Payload = {
		Prefix = Settings.Prefix,
		GetPlayerSeperator = Settings.GetPlayerSeperator,
		Options = Settings.Options
	}

	local success =
		pcall(function()
			writefile("Mada-Admin-V2/settings.json", Services.HttpService:JSONEncode(Payload))
		end)

	return success
end

function NFuncs.SafeHttpGet(URL, UseAsync)
	local RequestMethod = UseAsync and game.HttpGetAsync or game.HttpGet
	if type(RequestMethod) ~= "function" then
		return false, "HttpGet is not available in this executor"
	end

	return pcall(function()
		return RequestMethod(game, URL)
	end)
end

function NFuncs.ExecuteRemoteScript(URL, Label, UseAsync)
	if not loadstring then
		NotifyRuntime("Mada Admin", "This executor does not support loadstring for " .. tostring(Label), 5)
		return false
	end

	local success, Response = NFuncs.SafeHttpGet(URL, UseAsync)
	if not success or type(Response) ~= "string" or Response == "" then
		NotifyRuntime("Mada Admin", "Unable to download " .. tostring(Label), 5)
		return false
	end

	local compileSuccess, Compiled = pcall(loadstring, Response)
	if not compileSuccess or type(Compiled) ~= "function" then
		NotifyRuntime("Mada Admin", "Unable to compile " .. tostring(Label), 5)
		return false
	end

	local runSuccess, runErr = pcall(Compiled)
	if not runSuccess then
		warn("[Mada Admin] External script failed:", runErr)
		NotifyRuntime("Mada Admin", tostring(Label) .. " failed to run", 5)
		return false
	end

	return true
end

function NFuncs.ExecuteAssetScript(AssetId, Label)
	if not loadstring then
		NotifyRuntime("Mada Admin", "This executor does not support loadstring for " .. tostring(Label), 5)
		return false
	end

	local success, Objects = pcall(SafeGetObjects, AssetId)
	if not success or type(Objects) ~= "table" or not Objects[1] or type(Objects[1].Source) ~= "string" then
		NotifyRuntime("Mada Admin", "Unable to load asset for " .. tostring(Label), 5)
		return false
	end

	local compileSuccess, Compiled = pcall(loadstring, Objects[1].Source)
	if not compileSuccess or type(Compiled) ~= "function" then
		NotifyRuntime("Mada Admin", "Unable to compile asset for " .. tostring(Label), 5)
		return false
	end

	local runSuccess, runErr = pcall(Compiled)
	if not runSuccess then
		warn("[Mada Admin] Asset script failed:", runErr)
		NotifyRuntime("Mada Admin", tostring(Label) .. " failed to run", 5)
		return false
	end

	return true
end

function NFuncs.SafeJSONGet(URL)
	local success, Response = NFuncs.SafeHttpGet(URL, true)
	if not success or type(Response) ~= "string" then
		return false, Response
	end

	return pcall(function()
		return Services.HttpService:JSONDecode(Response)
	end)
end

function NFuncs.InstallNamecallProtection()
	if Runtime.Flags.NamecallProtectionInstalled then
		return true, Runtime.Flags.NamecallProtectionMode or "unknown"
	end

	local WrapClosure = newcclosure or function(Callback)
		return Callback
	end

	local function TryDirectMethodHooks()
		if not hookfunction then
			return false, "missing getnamecallmethod and hookfunction"
		end

		local HookedMethods = 0

		local function HookMethod(FunctionValue, ShouldBlock)
			if type(FunctionValue) ~= "function" then
				return false
			end

			local Original
			local success =
				pcall(function()
					Original =
						hookfunction(
							FunctionValue,
							WrapClosure(
								function(...)
									if ShouldBlock(...) then
										return
									end

									return Original(...)
								end
							)
						)
				end)

			if success and Original then
				HookedMethods = HookedMethods + 1
				return true
			end

			return false
		end

		HookMethod(
			Services.LocalPlayer and Services.LocalPlayer.Kick,
			function(Self, ...)
				return CommandToggles.AntiKick and Self == Services.LocalPlayer
			end
		)

		local function BlockTeleport(Self)
			return CommandToggles.AntiTeleport and Self == Services.TeleportService
		end

		HookMethod(Services.TeleportService and Services.TeleportService.Teleport, BlockTeleport)
		HookMethod(Services.TeleportService and Services.TeleportService.TeleportToPlaceInstance, BlockTeleport)
		HookMethod(Services.TeleportService and Services.TeleportService.TeleportAsync, BlockTeleport)

		if HookedMethods > 0 then
			Runtime.Flags.NamecallProtectionInstalled = true
			Runtime.Flags.NamecallProtectionMode = "directmethod"
			return true, "directmethod"
		end

		return false, "direct method hooks failed"
	end

	if not getnamecallmethod then
		return TryDirectMethodHooks()
	end

	local function BuildProtectedNamecall(OldNamecall)
		return WrapClosure(
			function(Self, ...)
				local Method = getnamecallmethod()
				local LowerMethod = tostring(Method):lower()

				if CommandToggles.AntiKick and Self == Services.LocalPlayer and LowerMethod == "kick" then
					return
				end

				if
					CommandToggles.AntiTeleport and
					(
						(Self == Services.TeleportService and LowerMethod == "teleport") or
						LowerMethod == "teleporttoplaceinstance"
					)
				then
					return
				end

				return OldNamecall(Self, ...)
			end
		)
	end

	if hookmetamethod then
		local OldNamecall
		local success =
			pcall(function()
				OldNamecall =
					hookmetamethod(
						game,
						"__namecall",
						BuildProtectedNamecall(function(Self, ...)
							return OldNamecall(Self, ...)
						end)
					)
			end)

		if success and OldNamecall then
			Runtime.Flags.NamecallProtectionInstalled = true
			Runtime.Flags.NamecallProtectionMode = "hookmetamethod"
			return true, "hookmetamethod"
		end
	end

	local SetReadOnly = setreadonly
	if not SetReadOnly and make_writeable and make_readonly then
		SetReadOnly =
			function(Table, ReadOnly)
				if ReadOnly then
					make_readonly(Table)
				else
					make_writeable(Table)
				end
			end
	end

	local GetRawMetatable = (debug and debug.getmetatable) or getrawmetatable
	if not (GetRawMetatable and SetReadOnly) then
		if not GetRawMetatable then
			return false, "missing getrawmetatable/debug.getmetatable"
		end
		return false, "missing setreadonly or make_writeable/make_readonly"
	end

	local GameMetaTable = GetRawMetatable(game)
	local OldNamecall = GameMetaTable and GameMetaTable.__namecall
	if type(OldNamecall) ~= "function" then
		return false, "__namecall unavailable on raw metatable"
	end

	local success =
		pcall(function()
			SetReadOnly(GameMetaTable, false)
			GameMetaTable.__namecall = BuildProtectedNamecall(OldNamecall)
			SetReadOnly(GameMetaTable, true)
		end)

	if not success then
		pcall(function()
			SetReadOnly(GameMetaTable, true)
		end)
		return false, "raw metatable is locked or write access failed"
	end

	Runtime.Flags.NamecallProtectionInstalled = true
	Runtime.Flags.NamecallProtectionMode = "rawmetatable"
	return true, "rawmetatable"
end

function NFuncs.CheckIfNPC(Character)
	if
		Character and Character.ClassName == "Model" and Character:FindFirstChildOfClass("Humanoid") and
		not Services.Players:GetPlayerFromCharacter(Character)
	then
		return true
	else
		return false
	end
end

function NFuncs.RandomString()
	local String = {}
	for St = 1, math.random(10, 15) do
		String[St] = string.char(math.random(32, 126))
	end
	return table.concat(String)
end

function NFuncs.BuildSearchQuery(...)
	local Args = {...}
	local QueryParts = {}
	for _, Value in next, Args do
		if Value and tostring(Value) ~= "" then
			table.insert(QueryParts, tostring(Value))
		end
	end
	return table.concat(QueryParts, " ")
end

function NFuncs.ParseArgumentsPreservingQuotes(Text)
	local Args = {}
	local Current = {}
	local InQuotes = false
	local QuoteChar = nil

	for i = 1, #Text do
		local Character = Text:sub(i, i)
		if (Character == "\"" or Character == "'") then
			if InQuotes and Character == QuoteChar then
				InQuotes = false
				QuoteChar = nil
			elseif not InQuotes then
				InQuotes = true
				QuoteChar = Character
			else
				table.insert(Current, Character)
			end
		elseif Character == " " and not InQuotes then
			if #Current > 0 then
				table.insert(Args, table.concat(Current))
				Current = {}
			end
		else
			table.insert(Current, Character)
		end
	end

	if #Current > 0 then
		table.insert(Args, table.concat(Current))
	end

	return Args
end

function PlayerFuncs.GetPlayer(Arg)
	local Target = {}
	local function AddUnique(Candidate)
		if Candidate and not table.find(Target, Candidate) then
			table.insert(Target, Candidate)
		end
	end

	local PlayerNames = Library.ParseText(Arg, Settings.GetPlayerSeperator)
	for i, String in next, PlayerNames or {"me"} do
		if String == nil then
			AddUnique(Services.LocalPlayer)
		elseif String:lower() == "random" then
			AddUnique(Services.Players:GetPlayers()[math.random(#Services.Players:GetPlayers())])
		elseif String:lower() == "me" then
			AddUnique(Services.LocalPlayer)
		elseif String:lower() == "all" then
			for i, Player in next, Services.Players:GetPlayers() do
				AddUnique(Player)
			end
		elseif String:lower() == "others" then
			for i, Player in next, Services.Players:GetPlayers() do
				if Player ~= Services.LocalPlayer then
					AddUnique(Player)
				end
			end
		elseif String:sub(1, 1) == "*" then --// Team search
			local TeamQuery = String:sub(2):lower()
			for i, Player in next, Services.Players:GetPlayers() do
				if TeamQuery ~= "" and tostring(Player.Team):lower():find(TeamQuery, 1, true) then
					AddUnique(Player)
				end
			end
		elseif String:lower() == "enemies" then
			for i, Player in next, Services.Players:GetPlayers() do
				if Player.Team ~= Services.LocalPlayer.Team then
					AddUnique(Player)
				end
			end
		elseif String:lower() == "dead" then
			for i, Player in next, Services.Players:GetPlayers() do
				if Player.Character and Player.Character.Humanoid then
					if Player.Character.Humanoid.Health == 0 then
						AddUnique(Player)
					end
				end
			end
		elseif String:lower() == "alive" then
			for i, Player in next, Services.Players:GetPlayers() do
				if Player.Character and Player.Character.Humanoid then
					if Player.Character.Humanoid.Health > 0 then
						AddUnique(Player)
					end
				end
			end
		elseif String:lower() == "npc" then
			for i, NPC in next, workspace:GetDescendants() do
				if NFuncs.CheckIfNPC(NPC) then
					AddUnique(NPC)
				end
			end
		elseif String:lower() == "friends" then
			for i, Player in next, Services.Players:GetPlayers() do
				if Player:IsFriendsWith(Services.LocalPlayer.UserId) and Player ~= Services.LocalPlayer then
					AddUnique(Player)
				end
			end
		elseif String:lower() == "nonfriends" then
			for i, Player in next, Services.Players:GetPlayers() do
				if not Player:IsFriendsWith(Services.LocalPlayer.UserId) and Player ~= Services.LocalPlayer then
					AddUnique(Player)
				end
			end
		elseif String:lower() == "closest" then
			local Lowest = math.huge
			local LPlayer
			if Services.LocalPlayer.Character then
				for i, Player in next, Services.Players:GetPlayers() do
					if Player.Character and Player ~= Services.LocalPlayer then
						local Distance =
							Player:DistanceFromCharacter(game.Players.LocalPlayer.Character.HumanoidRootPart.Position)
						if Distance < Lowest then
							Lowest = Distance
							LPlayer = Player
						end
					end
				end
				AddUnique(LPlayer)
			end
		elseif String:lower() == "farthest" then
			local Highest = 0
			local HPlayer
			if Services.LocalPlayer.Character then
				for i, Player in next, Services.Players:GetPlayers() do
					if Player.Character and Player ~= Services.LocalPlayer then
						local Distance = Player:DistanceFromCharacter(Services.Character.HumanoidRootPart.Position)
						if Distance > Highest then
							Highest = Distance
							HPlayer = Player
						end
					end
				end
				AddUnique(HPlayer)
			else
				return
			end
		else
			String = String:lower():gsub("%s", "")
			for _, Player in next, Services.Players:GetPlayers() do
				if Player.Name:lower():match(String) then
					AddUnique(Player)
				elseif Player.DisplayName:lower():match("^" .. String) then
					AddUnique(Player)
				end
			end
		end
	end
	return Target
end

function PlayerFuncs.Fling(Target)
	local Character = Services.LocalPlayer.Character
	if not Character or not Character.Parent then
		return false
	end

	local HumanoidRootPart = Character:FindFirstChild("HumanoidRootPart")
	local Humanoid = Character:FindFirstChildOfClass("Humanoid")
	if not HumanoidRootPart or not Humanoid then
		return false
	end

	local Targets = {}
	if typeof(Target) == "Instance" then
		table.insert(Targets, Target)
	elseif type(Target) == "table" then
		Targets = Target
	elseif type(Target) == "string" then
		Targets = PlayerFuncs.GetPlayer(Target)
	end

	local ValidTargets = {}
	for _, Candidate in next, Targets do
		if Candidate and Candidate ~= Services.LocalPlayer then
			local CandidateCharacter = Candidate.Character
			local CandidateRoot = CandidateCharacter and CandidateCharacter:FindFirstChild("HumanoidRootPart")
			local CandidateHumanoid = CandidateCharacter and CandidateCharacter:FindFirstChildOfClass("Humanoid")
			if CandidateRoot and CandidateHumanoid and CandidateHumanoid.Health > 0 then
				table.insert(ValidTargets, Candidate)
			end
		end
	end

	if #ValidTargets == 0 then
		return false
	end

	local OldCFrame = HumanoidRootPart.CFrame
	local OldVelocity = HumanoidRootPart.Velocity
	local OldRotVelocity = HumanoidRootPart.RotVelocity

	local Spinner = Instance.new("BodyAngularVelocity")
	Spinner.Name = "MadaFling"
	Spinner.MaxTorque = Vector3.new(9e9, 9e9, 9e9)
	Spinner.AngularVelocity = Vector3.new(0, 99999, 0)
	Spinner.P = 9e9
	Spinner.Parent = HumanoidRootPart

	for _, Candidate in next, ValidTargets do
		local CandidateRoot = Candidate.Character and Candidate.Character:FindFirstChild("HumanoidRootPart")
		if CandidateRoot then
			local StartTime = tick()
			repeat
				task.wait()
				HumanoidRootPart.CFrame =
					CandidateRoot.CFrame * CFrame.new(0, 0, 1.5) * CFrame.Angles(0, math.rad(math.random(0, 360)), 0)
				HumanoidRootPart.Velocity = Vector3.new(0, 999, 0)
				HumanoidRootPart.RotVelocity = Vector3.new(999, 99999, 999)
			until not CandidateRoot.Parent or tick() - StartTime > 0.55
		end
	end

	task.wait()
	Spinner:Destroy()
	HumanoidRootPart.CFrame = OldCFrame
	HumanoidRootPart.Velocity = OldVelocity
	HumanoidRootPart.RotVelocity = OldRotVelocity
	return true
end

function PlayerFuncs.UnFling()
	local Character = Services.LocalPlayer.Character
	if not Character or not Character.Parent then
		return false
	end

	local HumanoidRootPart = Character:FindFirstChild("HumanoidRootPart")
	if not HumanoidRootPart then
		return false
	end

	for _, Instance in next, HumanoidRootPart:GetChildren() do
		if Instance.Name == "Spin" or Instance.Name == "MadaFling" then
			Instance:Destroy()
		end
	end

	HumanoidRootPart.Velocity = Vector3.new(0, 0, 0)
	HumanoidRootPart.RotVelocity = Vector3.new(0, 0, 0)
	return true
end

function PlayerFuncs.Lay()
	local Character = Services.LocalPlayer.Character
	if not Character then
		Character = Services.LocalPlayer.CharacterAdded:Wait()
	end

	local Humanoid = Character:FindFirstChildWhichIsA("Humanoid")
	if not Humanoid then
		Character = Services.LocalPlayer.CharacterAdded:Wait()
		Humanoid = Character:FindFirstChildWhichIsA("Humanoid")
	end

	local RootPart = Humanoid and (Humanoid.RootPart or Character:FindFirstChild("HumanoidRootPart"))
	if not Humanoid or not RootPart then
		return false
	end

	-- Kolinski uses execCmd("unnosit") here; Mada has no nosit system to disable.
	Humanoid.Sit = true
	task.wait(0.1)
	RootPart.CFrame = RootPart.CFrame * CFrame.Angles(math.pi * 0.5, 0, 0)
	for _, Track in ipairs(Humanoid:GetPlayingAnimationTracks()) do
		Track:Stop()
	end
	return true
end

function PlayerFuncs.HeadSit(TargetName)
	local Targets = PlayerFuncs.GetPlayer(TargetName)
	local Target = nil
	for _, Candidate in next, Targets do
		if Candidate and Candidate ~= Services.LocalPlayer then
			Target = Candidate
			break
		end
	end

	if not Target then
		return false
	end

	CommandToggles.HeadSit = false
	task.wait(0.05)
	CommandToggles.HeadSit = true

	local JumpCancelled = false
	local JumpConnection
	JumpConnection =
		Services.UserInputService.JumpRequest:Connect(
			function()
				JumpCancelled = true
				CommandToggles.HeadSit = false
				if JumpConnection then
					JumpConnection:Disconnect()
					JumpConnection = nil
				end
			end
		)

	task.spawn(
		function()
			while CommandToggles.HeadSit do
				local Character = Services.LocalPlayer.Character
				local Humanoid = Character and Character:FindFirstChildWhichIsA("Humanoid")
				local Root = Character and Character:FindFirstChild("HumanoidRootPart")
				local TargetCharacter = Target.Character
				local TargetHead = TargetCharacter and TargetCharacter:FindFirstChild("Head")
				local TargetHumanoid = TargetCharacter and TargetCharacter:FindFirstChildWhichIsA("Humanoid")

				if not Character or not Root or not Humanoid or not TargetHead or not TargetHumanoid or TargetHumanoid.Health <= 0 then
					break
				end

				if JumpCancelled or Humanoid.Jump or Humanoid:GetState() == Enum.HumanoidStateType.Jumping then
					break
				end

				Humanoid.Sit = true
				Root.CFrame = TargetHead.CFrame * CFrame.new(0, 1.35, 0)
				Root.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
				Root.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
				task.wait()
			end

			CommandToggles.HeadSit = false
			if JumpConnection then
				JumpConnection:Disconnect()
				JumpConnection = nil
			end
		end
	)

	return true
end

function PlayerFuncs.Walkfling(Power)
	CommandToggles.Walkfling = false --// this is to stop the one before if the person is running touchfling again
	wait(0.3)
	CommandToggles.Walkfling = true
	spawn(
		function()
			local HumanoidRootPart, Character, Velocity, Movel = nil, nil, nil, 0.1
			repeat
				task.wait()
				if CommandToggles.Walkfling then
					while CommandToggles.Walkfling and
						not (Character and Character.Parent and HumanoidRootPart and HumanoidRootPart.Parent) do
						Services.RunService.Heartbeat:Wait()
						Character = Services.LocalPlayer.Character
						HumanoidRootPart = Character:FindFirstChild("HumanoidRootPart")
					end
					if CommandToggles.Walkfling then
						if
							unpack(PlayerFuncs.GetPlayer("closest")):DistanceFromCharacter(
							Services.Character.HumanoidRootPart.Position
							) <= 5
						then
							Velocity = HumanoidRootPart.Velocity
							HumanoidRootPart.Velocity = Velocity * tonumber(Power) + Vector3.new(0, tonumber(Power), 0)
							game:GetService("RunService").RenderStepped:Wait()
							if Character and Character.Parent and HumanoidRootPart and HumanoidRootPart.Parent then
								HumanoidRootPart.Velocity = Velocity
							end
							game:GetService("RunService").Stepped:Wait()
							if Character and Character.Parent and HumanoidRootPart and HumanoidRootPart.Parent then
								HumanoidRootPart.Velocity = Velocity + Vector3.new(0, Movel, 0)
								Movel = Movel * -1
							end
						end
					end
				end
			until not CommandToggles.Walkfling
		end
	)
end

		if Descendant:IsA("TextLabel") or Descendant:IsA("TextButton") or Descendant:IsA("TextBox") then
			UIFuncs.Tween(Descendant, TweenInformation, {TextTransparency = GoalTransparency})
		elseif Descendant:IsA("ImageLabel") or Descendant:IsA("ImageButton") then
			UIFuncs.Tween(Descendant, TweenInformation, {ImageTransparency = GoalTransparency})
		elseif Descendant:IsA("UIStroke") then

		elseif Descendant:IsA("Frame") then
			if Descendant.Name == "Outline" then
				UIFuncs.Tween(Descendant, TweenInformation, {BackgroundTransparency = GoalTransparency})
			end
		end
	end

function UIFuncs.Open(Frame, OffsetSizeX, OffsetSizeY, Speed)
	if Frame then
		Speed = Speed or UIFuncs.FrameSpeed
		Frame.Size = UDim2.new(0, 0, 0, 0)
		Frame.Visible = true
		Frame.AnchorPoint = Vector2.new(0.5, 0.5)
		if Frame.Position.X.Scale == 0 and Frame.Position.Y.Scale == 0 then
			Frame.Position = UDim2.new(0.5, 0, 0.5, 0)
		end

		local OpenTween =

		UIFuncs.Tween(Frame, OpenTween, {Size = UDim2.fromOffset(OffsetSizeX, OffsetSizeY)})
	end

		)
	end

function UIFuncs.Drag(Frame, Speed)
	local UserInputService = Services.UserInputService
	local RunService = (Services.RunService)
	local Dragging
	local DragInput
	local DragStart
	local StartPos

	local function Lerp(a, b, m)
		return a + (b - a) * m
	end

	if Speed == nil then
		Speed = 4
	end

	local LMPos
	local LGPos
	local DragSpeed = (tonumber(Speed))

	local function Update(dt)
		if not (StartPos) then
			return
		end
		if not (Dragging) and (LGPos) then
			Frame.Position =
				UDim2.new(
					StartPos.X.Scale,
					Lerp(Frame.Position.X.Offset, LGPos.X.Offset, dt * DragSpeed),
					StartPos.Y.Scale,
					Lerp(Frame.Position.Y.Offset, LGPos.Y.Offset, dt * DragSpeed)
				)
			return
		end

		local Delta = (LMPos - UserInputService:GetMouseLocation())
		local XG = (StartPos.X.Offset - Delta.X)
		local YG = (StartPos.Y.Offset - Delta.Y)
		LGPos = UDim2.new(StartPos.X.Scale, XG, StartPos.Y.Scale, YG)
		Frame.Position =
			UDim2.new(
				StartPos.X.Scale,
				Lerp(Frame.Position.X.Offset, XG, dt * DragSpeed),
				StartPos.Y.Scale,
				Lerp(Frame.Position.Y.Offset, YG, dt * DragSpeed)
			)
	end

	Frame.InputBegan:Connect(
		function(Input)
			if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
				Dragging = true
				DragStart = Input.Position
				StartPos = Frame.Position
				LMPos = UserInputService:GetMouseLocation()

				Input.Changed:Connect(
					function()
						if Input.UserInputState == Enum.UserInputState.End then
							Dragging = false
						end
					end
				)
			end
		end
	)

	Frame.InputChanged:Connect(
		function(Input)
			if
				Input.UserInputType == Enum.UserInputType.MouseMovement or
				Input.UserInputType == Enum.UserInputType.Touch
			then
				DragInput = Input
			end
		end
	)

	RunService.Heartbeat:Connect(Update)
end

		end
	)
end

function UIFuncs.GetCurrentTime()
	return tostring(os.date("%I") .. ":" .. os.date("%M") .. " " .. string.upper(tostring(os.date("%p"))))
end

function UIFuncs.SendNotification(Title, Description, Duration)
	spawn(
		function()
			Duration = Duration or 5
			local NotificationTemplate = UIFuncs.GetNotificationTemplate()
			if not NotificationTemplate then
				warn("Mada Admin: notification template not found")
				return
			end

			local ClonedNotif = NotificationTemplate:Clone()
			local TitleLabel = ClonedNotif:FindFirstChild("Title")
			local DescriptionLabel = ClonedNotif:FindFirstChild("Description")
			local Outline = ClonedNotif:FindFirstChild("Outline")
			local TimerFrame = ClonedNotif:FindFirstChild("Timer")
			local TimerBar = TimerFrame and TimerFrame:FindFirstChild("Timer")
			ClonedNotif.Parent = UIFuncs.NotificationHandler
			local OriginalPosition = ClonedNotif.Position
			local OriginalAnchorPoint = ClonedNotif.AnchorPoint
			ClonedNotif.Size = UDim2.new(0, 0, 0, 0)
			ClonedNotif.Position = OriginalPosition
			ClonedNotif.AnchorPoint = OriginalAnchorPoint
			ClonedNotif.BackgroundTransparency = 1
			if DescriptionLabel then
				DescriptionLabel.TextTransparency = 1
			end
			if TitleLabel then
				TitleLabel.TextTransparency = 1
			end
			if Outline then
				Outline.Transparency = 1
			end
			if TimerBar then
				TimerBar.Transparency = 0.25
			end
			ClonedNotif.Visible = true
			if DescriptionLabel then
				DescriptionLabel.Text = Description
			end
			if TitleLabel then
				TitleLabel.Text = Title
			end
			if TimerBar then
				TimerBar.Size = UDim2.new(0, 204, 0, 2)
			end
			UIFuncs.Tween(ClonedNotif, UIFuncs.SmoothStyles.Fade, {BackgroundTransparency = 0})
			if Outline then
				UIFuncs.Tween(Outline, UIFuncs.SmoothStyles.Fade, {Transparency = 0})
			end
			if TitleLabel then
				UIFuncs.Tween(TitleLabel, UIFuncs.SmoothStyles.Fade, {TextTransparency = 0})
			end
			if DescriptionLabel then
				UIFuncs.Tween(DescriptionLabel, UIFuncs.SmoothStyles.Fade, {TextTransparency = 0})
			end
			UIFuncs.Tween(ClonedNotif, UIFuncs.SmoothStyles.NotificationPop, {Size = UIFuncs.OriginalNotificationSize})
			if TimerBar then
				UIFuncs.Tween(
					TimerBar,
					TweenInfo.new(Duration, Enum.EasingStyle.Linear, Enum.EasingDirection.Out),
					{Size = UDim2.new(0, 1, 0, 2)}
				)
			end
			task.wait(Duration)
			if Outline then
				UIFuncs.Tween(Outline, UIFuncs.SmoothStyles.FadeFast, {Transparency = 1})
			end
			if TitleLabel then
				UIFuncs.Tween(TitleLabel, UIFuncs.SmoothStyles.FadeFast, {TextTransparency = 1})
			end
			if DescriptionLabel then
				UIFuncs.Tween(DescriptionLabel, UIFuncs.SmoothStyles.FadeFast, {TextTransparency = 1})
			end
			UIFuncs.Tween(ClonedNotif, UIFuncs.SmoothStyles.FadeFast, {BackgroundTransparency = 1})
			UIFuncs.Tween(ClonedNotif, UIFuncs.SmoothStyles.NotificationCollapse, {Size = UDim2.new(0, 0, 0, 40)})
			task.wait(math.max(0.14, UIFuncs.NotificationTweenSpeed) + 0.02)
			ClonedNotif:Destroy()
		end
	)
	task.wait(0.04)
end

	local OvershootSize = UDim2.fromOffset(FinalSize.X.Offset + 10, FinalSize.Y.Offset + 10)

		UDim2.new(FinalPosition.X.Scale, FinalPosition.X.Offset, FinalPosition.Y.Scale, FinalPosition.Y.Offset - 16)

	ButtonPulse.Visible = true
	ButtonPulse.Position = FinalPosition
	ButtonPulse.Size = UDim2.fromOffset(12, 12)
	ButtonPulse.BackgroundTransparency = 0.76
	Button.BackgroundTransparency = 1
	Button.TextTransparency = 1
	Button.TextStrokeTransparency = 1
	Button.Size = StartSize
	Button.Position = StartPosition
	Button.TextSize = math.max(10, FinalTextSize - 4)

	if ButtonStroke then
		ButtonStroke.Transparency = 1
		ButtonStroke.Thickness = 3
	end

	if ButtonGradient then
		ButtonGradient.Rotation = -110
	end

	if ButtonPulseStroke then
		ButtonPulseStroke.Transparency = 0.22
		ButtonPulseStroke.Thickness = 2.6
	end

	task.wait(0.06)

	UIFuncs.Tween(
		Button,
		TweenInfo.new(0.46, Enum.EasingStyle.Exponential, Enum.EasingDirection.Out),
		{
			Size = OvershootSize,
			Position = FinalPosition,
			BackgroundTransparency = 0,
			TextTransparency = 0
		}
	)
	UIFuncs.Tween(ButtonPulse, TweenInfo.new(0.52, Enum.EasingStyle.Exponential, Enum.EasingDirection.Out), {
		Size = UDim2.fromOffset(FinalSize.X.Offset + 54, FinalSize.Y.Offset + 54),
		BackgroundTransparency = 1
	})

	if ButtonStroke then
		UIFuncs.Tween(ButtonStroke, TweenInfo.new(0.34, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {
			Transparency = 0.06,
			Thickness = 1.9
		})

			Transparency = 1,

		})

			Rotation = 28
		})

function UIFuncs.LoadTheme()
	local Theme = Settings.Themes

	for i, Child in next, Interface:GetDescendants() do
		if Child:IsA("Frame") and Child.Parent == Interface then
			Child.BackgroundColor3 = Theme.CardColor
		elseif Child.Name == "Example" and Child:IsA("Frame") then
			Child.BackgroundColor3 = Theme.CardColorAlt
		elseif Child:IsA("ImageLabel") or Child:IsA("ImageButton") then
			if Child.Name == "Edge" then
				Child.ImageColor3 = Theme.EdgeColor
			else
				Child.ImageColor3 = Theme.ImageColor
				Child.BackgroundColor3 = Theme.CardColor
			end
		elseif Child:IsA("CanvasGroup") then
			Child.BackgroundColor3 = Theme.CardColorAlt
		elseif Child.Name == "Description" and Child:IsA("TextLabel") then
			Child.TextColor3 = Theme.SecondaryTextColor
			Child.Font = Enum.Font.GothamMedium
		elseif (Child.Name == "Topbar" or Child.Name == "Bottom") and Child:IsA("Frame") then
			Child.BackgroundColor3 = Theme.TopbarColor
		elseif Child:IsA("TextButton") or Child:IsA("TextBox") then
			if Child.Name == "Input" and Child.Parent.Name == "CenterBar" then
				Child.BackgroundColor3 = Theme.CommandCenterBarColor
				Child.TextColor3 = Theme.InputTextColor
				Child.PlaceholderColor3 = Theme.SecondaryTextColor
				Child.Font = Enum.Font.GothamSemibold
			else
				Child.BackgroundColor3 = Theme.CardColorAlt
				Child.TextColor3 = Theme.PrimaryTextColor
				if Child:IsA("TextButton") then
					Child.Font = Enum.Font.GothamSemibold
				end
			end
		elseif Child.Name == "Title" and Child:IsA("TextLabel") then
			Child.TextColor3 = Theme.PrimaryTextColor
			Child.Font = Enum.Font.GothamBold
		elseif Child:IsA("TextLabel") then
			Child.TextColor3 = Theme.PrimaryTextColor
			if Child.Name ~= "Description" then
				Child.Font = Enum.Font.GothamSemibold
			end
		elseif Child.Name == "Bottom1" and Child.Parent.Name == "Topbar" then
			Child.BackgroundColor3 = Theme.TopbarColor
		elseif Child:IsA("Frame") and Child.Name == "Outline" then
			Child.BackgroundColor3 = Theme.OutlineColor2
		elseif Child:IsA("UIStroke") then
			Child.Color = Theme.OutlineColor2
		elseif Child:IsA("Frame") and Child.Name == "Horizontal" then
			if Child.Parent.Name ~= "CenterBar" then
				Child.BackgroundColor3 = Theme.CommandLinesColor
			elseif Child:IsA("Frame") and Child.Parent.Name == "Background" then
				Child.BackgroundColor3 = Theme.AutofillColor
			else
				Child.BackgroundColor3 = Theme.CommandCenterBarColor
			end
		elseif Child.Name == "Input" and Child:IsA("TextLabel") and Child.Parent:FindFirstChild("Background") then
			Child.TextColor3 = Theme.AutofillTextColor
		elseif Child:IsA("ScrollingFrame") then
			Child.ScrollBarImageColor3 = Theme.AccentColor
			Child.ScrollBarThickness = 4
		end
	end

	if Interface and Interface:FindFirstChild("CmdBar") and Interface.CmdBar:FindFirstChild("CenterBar") then
		local Center = Interface.CmdBar.CenterBar
		local Input = Center:FindFirstChild("Input")
		if Input and Input:IsA("TextBox") then
			Input.BackgroundColor3 = Theme.CommandCenterBarColor
			Input.BackgroundTransparency = 0
			Input.TextColor3 = Theme.InputTextColor
			Input.PlaceholderColor3 = Theme.SecondaryTextColor
			Input.TextTransparency = 0
			Input.TextStrokeTransparency = 1
			Input.ClearTextOnFocus = false
			Input.RichText = false
			Input.TextScaled = false
			Input.TextWrapped = false
			Input.TextSize = 18
			Input.Font = Enum.Font.GothamSemibold
			Input.TextXAlignment = Enum.TextXAlignment.Center
			Input.TextYAlignment = Enum.TextYAlignment.Center
			Input.ZIndex = math.max(Center.ZIndex + 2, 8)
		end
	end

	local NotificationTemplate = UIFuncs.GetNotificationTemplate()
	if NotificationTemplate then
		NotificationTemplate.BackgroundColor3 = Theme.CardColor
	end
	Interface.Commands.Tabs.CommandInfoExample.Arguments.BackgroundColor3 = Theme.CardColorAlt
	Button.BackgroundColor3 = Theme.CardColor
	Button.TextColor3 = Theme.PrimaryTextColor
end

function UIFuncs.ApplyModernStyling()
	local Theme = Settings.Themes

	local function EnsureCorner(Object, Radius)
		if not Object:FindFirstChild("MadaCorner") then
			UIFuncs.CreateInstance(
				"UICorner",
				{
					Name = "MadaCorner",
					CornerRadius = Radius or UDim.new(0, 12),
					Parent = Object
				}
			)
		end
	end

	local function EnsureStroke(Object, Thickness, Transparency, Color)
		local Stroke = Object:FindFirstChild("MadaStroke")
		if not Stroke then
			Stroke =
				UIFuncs.CreateInstance(
					"UIStroke",
					{
						Name = "MadaStroke",
						ApplyStrokeMode = Enum.ApplyStrokeMode.Border,
						Parent = Object
					}
				)
		end

		Stroke.Thickness = Thickness or 1
		Stroke.Transparency = Transparency or 0.35
		Stroke.Color = Color or Theme.OutlineColor2
	end

	local function EnsureFill(Object, Name, ColorA, ColorB, Rotation)
		local Fill = Object:FindFirstChild(Name)
		if not Fill then
			Fill =
				UIFuncs.CreateInstance(
					"UIGradient",
					{
						Name = Name,
						Parent = Object
					}
				)
		end

		Fill.Color =
			ColorSequence.new({
				ColorSequenceKeypoint.new(0, ColorA),
				ColorSequenceKeypoint.new(0.5, ColorB),
				ColorSequenceKeypoint.new(1, ColorA:Lerp(ColorB, 0.3))
			})
		Fill.Rotation = Rotation or 90
	end

	local function EnsureAccentBar(Object, Name, ColorA, ColorB)
		local Bar = Object:FindFirstChild(Name)
		if not Bar then
			Bar =
				UIFuncs.CreateInstance(
					"Frame",
					{
						Name = Name,
						AnchorPoint = Vector2.new(0.5, 0),
						Position = UDim2.new(0.5, 0, 0, 0),
						Size = UDim2.new(1, -24, 0, 4),
						BorderSizePixel = 0,
						ZIndex = (Object.ZIndex or 1) + 1,
						Parent = Object
					}
				)
			UIFuncs.CreateInstance("UICorner", {CornerRadius = UDim.new(1, 0), Parent = Bar})
		end

		Bar.BackgroundColor3 = ColorA
		EnsureFill(Bar, "MadaAccentGradient", ColorA, ColorB, 0)
	end

	local function EnsureBackdrop()
		local Backdrop = Interface:FindFirstChild("MadaBackdrop")
		if not Backdrop then
			Backdrop =
				UIFuncs.CreateInstance(
					"Frame",
					{
						Name = "MadaBackdrop",
						Size = UDim2.fromScale(1, 1),
						BackgroundColor3 = Theme.OverlayColor,
						BackgroundTransparency = 1,
						BorderSizePixel = 0,
						ZIndex = -10,
						Parent = Interface
					}
				)

			local GlowA =
				UIFuncs.CreateInstance(
					"Frame",
					{
						Name = "GlowA",
						AnchorPoint = Vector2.new(0.5, 0.5),
						Position = UDim2.new(-0.06, 0, 0.05, 0),
						Size = UDim2.fromOffset(180, 180),
						BackgroundColor3 = Theme.AccentColor,
						BackgroundTransparency = 0.93,
						BorderSizePixel = 0,
						ZIndex = -9,
						Parent = Backdrop
					}
				)

			local GlowB =
				UIFuncs.CreateInstance(
					"Frame",
					{
						Name = "GlowB",
						AnchorPoint = Vector2.new(0.5, 0.5),
						Position = UDim2.new(1.06, 0, 0.94, 0),
						Size = UDim2.fromOffset(220, 220),
						BackgroundColor3 = Theme.AccentColorDark,
						BackgroundTransparency = 0.95,
						BorderSizePixel = 0,
						ZIndex = -9,
						Parent = Backdrop
					}
				)

			UIFuncs.CreateInstance("UICorner", {CornerRadius = UDim.new(1, 0), Parent = GlowA})
			UIFuncs.CreateInstance("UICorner", {CornerRadius = UDim.new(1, 0), Parent = GlowB})
			EnsureFill(Backdrop, "MadaBackdropGradient", Theme.OverlayColor, Theme.PrimaryBGColor, 125)
		end

		Backdrop.BackgroundColor3 = Theme.OverlayColor
		Backdrop.BackgroundTransparency = 1

		local GlowA = Backdrop:FindFirstChild("GlowA")
		if GlowA and GlowA:IsA("Frame") then
			GlowA.Visible = false
			GlowA.BackgroundTransparency = 1
		end

		local GlowB = Backdrop:FindFirstChild("GlowB")
		if GlowB and GlowB:IsA("Frame") then
			GlowB.Visible = false
			GlowB.BackgroundTransparency = 1
		end
	end

	local function EnsureGradient(Object, Name, ColorA, ColorB, Rotation)
		local Gradient = Object:FindFirstChild(Name)
		if not Gradient then
			Gradient =
				UIFuncs.CreateInstance(
					"UIGradient",
					{
						Name = Name,
						Parent = Object
					}
				)
		end

		Gradient.Color =
			ColorSequence.new({
				ColorSequenceKeypoint.new(0, ColorA),
				ColorSequenceKeypoint.new(1, ColorB)
			})
		Gradient.Rotation = Rotation or 90
	end

	local function RemoveGradient(Object, Name)
		local Gradient = Object:FindFirstChild(Name)
		if Gradient and Gradient:IsA("UIGradient") then
			Gradient:Destroy()
		end
	end

	local function StyleReadableButton(ButtonObject)
		if not ButtonObject or not ButtonObject:IsA("TextButton") then
			return
		end

		RemoveGradient(ButtonObject, "MadaButtonGradient")
		ButtonObject.BackgroundColor3 = Theme.CardColorAlt
		ButtonObject.TextColor3 = Theme.PrimaryTextColor
		ButtonObject.TextTransparency = 0

		for _, Descendant in next, ButtonObject:GetDescendants() do
			if Descendant:IsA("TextLabel") then
				if Descendant.Name == "Description" then
					Descendant.TextColor3 = Theme.SecondaryTextColor
				else
					Descendant.TextColor3 = Theme.PrimaryTextColor
				end
				Descendant.TextTransparency = 0
			elseif Descendant:IsA("ImageLabel") or Descendant:IsA("ImageButton") then
				Descendant.ImageColor3 = Theme.ImageColor
				Descendant.ImageTransparency = 0
			end
		end
	end

	EnsureBackdrop()

	for _, Panel in next, {Interface.Commands, Interface.Options, Interface.Logs, Interface.Servers, Interface.Introduction} do
		if Panel then
			Panel.BackgroundColor3 = Theme.CardColor
			EnsureCorner(Panel, UDim.new(0, 22))
			EnsureStroke(Panel, 1.4, 0.12, Theme.OutlineColor2)
			EnsureGradient(Panel, "MadaPanelGradient", Theme.CardColor, Theme.PrimaryBGColor, 130)
			EnsureAccentBar(Panel, "MadaPanelAccent", Theme.AccentColor, Theme.AccentSoft)
		end
	end

	for _, Topbar in next, {
		Interface.Commands:FindFirstChild("Topbar"),
		Interface.Options:FindFirstChild("Topbar"),
		Interface.Logs:FindFirstChild("Topbar"),
		Interface.Servers:FindFirstChild("Topbar"),
		Interface.Introduction:FindFirstChild("Topbar")
	} do
		if Topbar then
			Topbar.BackgroundColor3 = Theme.TopbarColor
			EnsureGradient(Topbar, "MadaTopbarGradient", Theme.TopbarColor, Theme.CardColorAlt, 0)
			EnsureStroke(Topbar, 1.2, 0.22, Theme.AccentSoft)
		end
	end

	for _, Clickable in next, Interface:GetDescendants() do
		if Clickable:IsA("TextButton") then
			EnsureCorner(Clickable, UDim.new(0, 12))
			if Clickable ~= Button then
				EnsureStroke(Clickable, 1.1, 0.28, Theme.OutlineColor2)
				if
					Interface.Options and Interface.Options:FindFirstChild("Tabs") and
						Clickable:IsDescendantOf(Interface.Options.Tabs)
				then
					StyleReadableButton(Clickable)
				else
					EnsureGradient(Clickable, "MadaButtonGradient", Theme.CardColorAlt, Theme.TopbarColor, 35)
				end
			end
		elseif Clickable:IsA("TextBox") then
			EnsureCorner(Clickable, UDim.new(0, 14))
			EnsureStroke(Clickable, 1.2, 0.24, Theme.OutlineColor2)
			RemoveGradient(Clickable, "MadaInputGradient")
			Clickable.BackgroundTransparency = 0
		elseif Clickable:IsA("ScrollingFrame") then
			EnsureCorner(Clickable, UDim.new(0, 16))
			EnsureStroke(Clickable, 1, 0.42, Theme.OutlineColor2)
		elseif Clickable:IsA("Frame") and Clickable.Name == "Command" then
			EnsureCorner(Clickable, UDim.new(0, 16))
			EnsureStroke(Clickable, 1, 0.22, Theme.OutlineColor2)
			EnsureGradient(Clickable, "MadaCommandGradient", Theme.CardColorAlt, Theme.CardColor, 18)
			EnsureAccentBar(Clickable, "MadaCommandAccent", Theme.AccentColorDark, Theme.AccentColor)
		end
	end

	local NotificationTemplate = UIFuncs.GetNotificationTemplate()
	if NotificationTemplate then
		EnsureCorner(NotificationTemplate, UDim.new(0, 16))
		EnsureStroke(NotificationTemplate, 1.1, 0.12, Theme.AccentColor)
		EnsureGradient(NotificationTemplate, "MadaNotificationGradient", Theme.CardColorAlt, Theme.CardColor, 20)
	end

	if Popup then
		EnsureCorner(Popup, UDim.new(0, 20))
		EnsureStroke(Popup, 1.2, 0.1, Theme.AccentColor)
		EnsureGradient(Popup, "MadaPopupGradient", Theme.CardColor, Theme.CardColorAlt, 140)
		if Popup:FindFirstChild("Bottom") then
			EnsureGradient(Popup.Bottom, "MadaPopupBottomGradient", Theme.TopbarColor, Theme.CardColorAlt, 0)
		end
	end

	if Interface and Interface:FindFirstChild("Top") then
		local TopFrame = Interface.Top
		TopFrame.BackgroundColor3 = Theme.CardColorAlt
		EnsureCorner(TopFrame, UDim.new(0, 18))
		EnsureStroke(TopFrame, 1.15, 0.16, Theme.AccentSoft)
		EnsureGradient(TopFrame, "MadaQuickBarGradient", Theme.CardColorAlt, Theme.TopbarColor, 0)
	end

	if Interface and Interface:FindFirstChild("CmdBar") then
		local CmdBar = Interface.CmdBar
		local Center = CmdBar:FindFirstChild("CenterBar")
		local Left = CmdBar:FindFirstChild("LeftFill")

local CommandAmount = 0
local Command = {}
local Command2 = {}
local Commands, Aliases = {}, {}
local LastRanCommand
local CommandHistory = {}
Command.Add = function(...)
	local vars = {...}
	local aliases, info, arguments, func = vars[1], vars[2], vars[3], vars[4]
	for i, CommandName in next, aliases do
		if i == 1 then
			Commands[CommandName:lower()] = {func, info, arguments}
		else
			Aliases[CommandName:lower()] = {func, info, arguments}
		end
	end
	CommandAmount = CommandAmount + 1
end

Command2.Add = function(...)
	local vars = {...}
	local aliases, info, arguments, func = vars[1], vars[2], vars[3], vars[4]
	for i, CommandName in next, aliases do
		if i == 1 then
			Command2[CommandName:lower()] = {func, info, arguments}
		else
			Aliases[CommandName:lower()] = {func, info, arguments}
		end
	end
	CommandAmount = CommandAmount + 1
end

Command.Run = function(Arguments, PreserveLastCommand)
	if type(Arguments) ~= "table" then
		return
	end

	local arguments = NFuncs.ShallowCopy(Arguments)
	if arguments[1] then
		if not PreserveLastCommand then
			LastRanCommand = NFuncs.ShallowCopy(arguments)
			table.insert(CommandHistory, NFuncs.ShallowCopy(arguments))
			if #CommandHistory > 25 then
				table.remove(CommandHistory, 1)
			end
		end

		local caller = arguments[1]
		table.remove(arguments, 1)
		local success, msg =
			pcall(
				function()
					if Commands[caller:lower()] then
						Commands[caller:lower()][1](unpack(arguments))
				elseif Aliases[caller:lower()] then
						Aliases[caller:lower()][1](unpack(arguments))
				else
						UIFuncs.SendNotification(
							"Mada Admin",
							"The command <b>" .. tostring(caller) .. "</b> does not exist!",
							5
						)
					end
				end
			)

		if not success then
			UIFuncs.SendNotification(
				"Command Error!",
				"This command has had an error so it could not run, please check your <b>console</b> for more information",
				5
			)
			warn(msg)
		end
	end
end

Library.Find = function(T, V)

			return i

Library.ParseText = function(Text, Watch)
	local Parsed = {}
	if not Text then
		return nil
	end
	for Arguments in Text:gmatch("[^" .. Watch .. "]+") do
		Arguments = Arguments:gsub("-", "%%-")
		local pos = Text:find(Arguments)
		Arguments = Arguments:gsub("%%", "")
		if pos then
			local find = Text:sub(pos - Settings.Prefix:len(), pos - 1)
			if (find == Settings.Prefix and Watch == Settings.Prefix) or Watch ~= Settings.Prefix then
				table.insert(Parsed, Arguments)
			end
		else
			table.insert(Parsed, nil)
		end
	end
	return Parsed
end

Library.ParseCommand = function(Text)
	spawn(

			local commands = Library.ParseText(Text, Settings.Prefix)
			for _, parsed in next, commands do
				local args = NFuncs.ParseArgumentsPreservingQuotes(parsed)
				Command.Run(args)
			end
		end
	)
end

function UIFuncs.ResolveOptionsTab(Tabs, ButtonObject)
	if not Tabs or not ButtonObject then
		return nil
	end

	local function Normalize(Text)
		return tostring(Text or ""):lower():gsub("%s+", ""):gsub("[^%w]", "")
	end

	local function ExtractButtonLabel(Object)
		local Candidates = {}

		if Object:IsA("TextButton") or Object:IsA("TextLabel") or Object:IsA("TextBox") then
			table.insert(Candidates, Object.Text or "")
		end

		for _, Descendant in next, Object:GetDescendants() do
			if Descendant:IsA("TextLabel") or Descendant:IsA("TextButton") or Descendant:IsA("TextBox") then
				if Descendant.Text and Descendant.Text ~= "" then
					table.insert(Candidates, Descendant.Text)
				end
			end
		end

		for _, Candidate in next, Candidates do
			local Normalized = Normalize(Candidate)
			if Normalized ~= "" then
				return Normalized
			end
		end

		return Normalize(ButtonObject.Name)
	end

	local function FindTabByAliases(Aliases)
		for _, Child in next, Tabs:GetChildren() do
			if Child:IsA("ScrollingFrame") and Child.Name ~= "Selection" then
				local TabName = Normalize(Child.Name)
				for _, Alias in next, Aliases do
					if TabName == Alias or TabName:find(Alias, 1, true) then
						return Child
					end
				end
			end
		end
		return nil
	end

	local Label = ExtractButtonLabel(ButtonObject)
	if Label == "information" or Label == "info" then
		return FindTabByAliases({"information", "info"})
	elseif Label == "waypoints" or Label == "waypoint" then
		return FindTabByAliases({"waypoints", "waypoint"})
	elseif Label == "plugins" or Label == "plugin" then
		return FindTabByAliases({"plugins", "plugin"})
	elseif Label == "updates" or Label == "update" or Label == "versions" or Label == "version" then
		return FindTabByAliases({"updates", "update", "versions", "version"})
	end

	return nil
end

local Speed = 3
local function onUpdate(dt)
	pcall(
		function()
			if (isFlying) then
				local cf = Services.Camera.CFrame
				local direction =
					cf.rightVector * (Movement.right - Movement.left) +
					cf.lookVector * (Movement.forward - Movement.backward)

				if (direction:Dot(direction) > 0) then
					direction = direction.unit
				end

				BodyGyro.CFrame = cf
				BodyVelocity.Velocity = direction * Services.Character.Humanoid.WalkSpeed * Speed
			end
		end
	)
end

local function ModifyMovement(newMovement)
	Movement = newMovement or Movement
	if (isFlying) then

	end
end

local Controller = require(Services.LocalPlayer:FindFirstChild("ControlModule", true))
local TouchFrame = Services.LocalPlayer:FindFirstChild("TouchControlFrame", true)

if Controller and TouchFrame then
	local IsMovingThumbstick = false
	local DeadZone = 0.15
	local DeadZoneNormalized = 1 - DeadZone

	local function isTouchOnThumbstick(Position)
		if not TouchFrame then
			return false
		end
		local ClassicFrame = TouchFrame:FindFirstChild("ThumbstickFrame")
		local DynamicFrame = TouchFrame:FindFirstChild("DynamicThumbstickFrame")

		local StickFrame = (ClassicFrame and ClassicFrame.Visible) and ClassicFrame or DynamicFrame
		if (StickFrame) then
			local StickPosition = StickFrame.AbsolutePosition
			local StickSize = StickFrame.AbsoluteSize
			return Position.X >= StickPosition.X and Position.X <= (StickPosition.X + StickSize.X) and
				Position.Y >= StickPosition.Y and
				Position.Y <= (StickPosition.Y + StickSize.Y)
		end
		return false
	end

	UIS.TouchStarted:Connect(
		function(touch, gameProcessedEvent)
			isMovingThumbstick = isTouchOnThumbstick(touch.Position)
		end
	)

	UIS.TouchEnded:Connect(
		function(touch, gameProcessedEvent)
			if not isMovingThumbstick then
				return
			end
			isMovingThumbstick = false
			ModifyMovement({forward = 0, backward = 0, right = 0, left = 0})
		end
	)

	UIS.TouchMoved:Connect(
		function(touch, gameProcessedEvent)
			if not isMovingThumbstick then
				return
			end
			local MouseVector = Controller:GetMoveVector()
			local LeftRight = MouseVector.X
			local ForeBack = MouseVector.Z

			-- change the range from [0.15-1] to [0-1] if it is outside the deadzone
			Movement.left = LeftRight < -DeadZone and -(LeftRight - DeadZone) / DeadZoneNormalized or 0
			Movement.right = LeftRight > DeadZone and (LeftRight - DeadZone) / DeadZoneNormalized or 0

			Movement.forward = ForeBack < -DeadZone and -(ForeBack - DeadZone) / DeadZoneNormalized or 0
			Movement.backward = ForeBack > DeadZone and (ForeBack - DeadZone) / DeadZoneNormalized or 0
			ModifyMovement()
		end
	)
end

Command.Add(
	{"BunnyHop", "BHOP"},
	{"BunnyHop / BHOP", "Bunny hop"},
	{"None"},
	function()
		task.spawn(function()	 
			function Init(Player, Camera, Input)
				collider = Services.Character.HumanoidRootPart
				camera = Camera
				input = Input
				playerVelocity = 0
				playerGrounded = false
				moveInputSum = {
					["forward"] = 0,
					["side"] 	= 0 --left is positive
				}

				airAccelerate 			= 10000
				airMaxSpeed 			= 2.4
				groundAccelerate 		= 250
				groundMaxVelocity 		= 20
				friction			 	= 10
				playerTorsoToGround 	= 3
				movementStickDistance 	= 0.5
				jumpVelocity 			= 52.5
				movementPositionForce	= 400000
				movementVelocityForce	= 300000
				maxMovementPitch		= 0.6
				rayYLength				= playerTorsoToGround + movementStickDistance
				movementPositionD		= 125
				movementPositionP		= 14000
				movementVelocityP		= 1500
				gravity					= 0.4

			end

			function InitBodyMovers()
				movementPosition = Instance.new("BodyPosition", collider)
				movementPosition.Name = "movementPosition"
				movementPosition.D = movementPositionD
				movementPosition.P = movementPositionP
				movementPosition.maxForce = Vector3.new()
				movementPosition.position = Vector3.new()

				movementVelocity = Instance.new("BodyVelocity", collider)
				movementVelocity.Name = "movementVelocity"
				movementVelocity.P = movementVelocityP
				movementVelocity.maxForce = Vector3.new()
				movementVelocity.velocity = Vector3.new()

				gravityForce = Instance.new("BodyForce", collider)
				gravityForce.Name = "gravityForce"
				gravityForce.force = Vector3.new(0, (1-gravity)*196.2, 0) * GetCharacterMass()
			end

			function update(deltaTime)
				DT = deltaTime
				UpdateMoveInputSum()
				CameraYaw = GetYaw()
				CameraLook = CameraYaw.lookVector	
				if CameraLook == nil then
					return
				end
				local HitPart, HitPosition, HitNormal, YRatio, ZRatio = FindCollisionRay()
				partYRatio = YRatio
				partZRatio = ZRatio

				playerGrounded = HitPart ~= nil and true or false
				playerVelocity = collider.Velocity - Vector3.new(0, collider.Velocity.y, 0)
				if playerGrounded and (input["Space"] or jumping) then
					jumping = true
				else
					jumping = false
				end

				setCharacterRotation()
				if jumping then
					jump()
				elseif playerGrounded then
					run(HitPosition)
				else
					air()		
				end

			end

			function UpdateMoveInputSum()
				moveInputSum["forward"] = input["W"] == true and 1 or 0
				moveInputSum["forward"] = input["S"] == true and moveInputSum["forward"] - 1 or moveInputSum["forward"]
				moveInputSum["side"] = input["A"] == true and 1 or 0
				moveInputSum["side"] = input["D"] == true and moveInputSum["side"] - 1 or moveInputSum["side"]
			end

			function FindCollisionRay()
				local torsoCFrame = Services.Character.HumanoidRootPart.CFrame
				local ignoreList = {Services.Character, camera}
				local rays = {
					Ray.new(Services.Character.HumanoidRootPart.Position, Vector3.new(0, -rayYLength, 0)),
					Ray.new((torsoCFrame * CFrame.new(-0.8,0,0)).p, Vector3.new(0, -rayYLength, 0)),
					Ray.new((torsoCFrame * CFrame.new(0.8,0,0)).p, Vector3.new(0, -rayYLength, 0)),
					Ray.new((torsoCFrame * CFrame.new(0,0,0.8)).p, Vector3.new(0, -rayYLength, 0)),
					Ray.new((torsoCFrame * CFrame.new(0,0,-0.8)).p, Vector3.new(0, -rayYLength, 0))
				}
				local rayReturns  = {}

				local i
				for i = 1, #rays do
					local part, position, normal = game.Workspace:FindPartOnRayWithIgnoreList(rays[i],ignoreList)
					if part == nil then
						position = Vector3.new(0,-3000000,0)
					end
					if i == 1 then
						table.insert(rayReturns, {part, position, normal})
					else
						local yPos = position.y
						if yPos <= rayReturns[#rayReturns][2].y then
							table.insert(rayReturns, {part, position, normal})
						else 
							local j
							for j = 1, #rayReturns do
								if yPos >= rayReturns[j][2].y then
									table.insert(rayReturns, j, {part, position, normal})
								end
							end
						end
					end
				end

				i = 1
				local YRatio, ZRatio = getPartYRatio(rayReturns[i][3])
				while Magnitude2D(YRatio, ZRatio) > maxMovementPitch and i<#rayReturns do
					i = i + 1
					if rayReturns[i][1] then
						YRatio, ZRatio = getPartYRatio(rayReturns[i][3])
					end
				end

				return rayReturns[i][1], rayReturns[i][2], rayReturns[i][3], YRatio, ZRatio
			end

			function setCharacterRotation()
				local rotationLook = collider.Position + camera.CoordinateFrame.lookVector
				collider.CFrame = CFrame.new(collider.Position, Vector3.new(rotationLook.x, collider.Position.y, rotationLook.z))
				collider.RotVelocity = Vector3.new()
			end

			function jump()
				collider.Velocity = Vector3.new(collider.Velocity.x, jumpVelocity, collider.Velocity.z)
				air()
			end

			function air()
				movementPosition.maxForce = Vector3.new()
				movementVelocity.velocity = getMovementVelocity(collider.Velocity, airAccelerate, airMaxSpeed)
				movementVelocity.maxForce = getMovementVelocityAirForce()
			end

			function run(HitPosition)
				local playerSpeed = collider.Velocity.magnitude
				local mVelocity = collider.Velocity

				if playerSpeed ~= 0 then
					local drop = playerSpeed * friction * DT;
					mVelocity = mVelocity * math.max(playerSpeed - drop, 0) / playerSpeed;
				end

				movementPosition.position = HitPosition + Vector3.new(0,playerTorsoToGround,0)
				movementPosition.maxForce = Vector3.new(0,movementPositionForce,0)
				movementVelocity.velocity = getMovementVelocity(mVelocity, groundAccelerate, groundMaxVelocity)
				local VelocityForce = getMovementVelocityForce()
				movementVelocity.maxForce = VelocityForce
				movementVelocity.P = movementVelocityP
			end

			function getMovementVelocity(prevVelocity, accelerate, maxVelocity)
				local accelForward = CameraLook * moveInputSum["forward"]
				local accelSide = (CameraYaw * CFrame.Angles(0,math.rad(90),0)).lookVector * moveInputSum["side"];
				local accelDir = (accelForward+accelSide).unit;
				if moveInputSum["forward"] == 0 and moveInputSum["side"] == 0 then --avoids divide 0 errors
					accelDir = Vector3.new(0,0,0);
				end

				local projVel =  prevVelocity:Dot(accelDir);
				local accelVel = accelerate * DT;

				if (projVel + accelVel > maxVelocity) then
					accelVel = math.max(maxVelocity - projVel, 0);
				end

				return prevVelocity + accelDir * accelVel;
			end

			function getMovementVelocityForce()

				return Vector3.new(movementVelocityForce,0,movementVelocityForce)
			end

			function getMovementVelocityAirForce()
				local accelForward = CameraLook * moveInputSum["forward"];
				local accelSide = (CameraYaw * CFrame.Angles(0,math.rad(90),0)).lookVector * moveInputSum["side"]
				local accelDir = (accelForward+accelSide).unit
				if moveInputSum["forward"] == 0 and moveInputSum["side"] == 0 then
					accelDir = Vector3.new(0,0,0);
				end

				local xp = math.abs(accelDir.x)
				local zp = math.abs(accelDir.z)

				return Vector3.new(movementVelocityForce*xp,0,movementVelocityForce*zp)
			end

			function getPartYRatio(normal)
				local partYawVector = Vector3.new(-normal.x, 0, -normal.z)
				if partYawVector.magnitude == 0 then
					return 0,0
				else
					local partPitch = math.atan2(partYawVector.magnitude,normal.y)/(math.pi/2)
					local vector = Vector3.new(CameraLook.x, 0, CameraLook.z)*partPitch
					return vector:Dot(partYawVector), -partYawVector:Cross(vector).y
				end	
			end

			function GetYaw() --returns CFrame
				return camera.CoordinateFrame*CFrame.Angles(-GetPitch(),0,0)
			end

			function GetPitch() --returns number
				return math.pi/2 - math.acos(camera.CoordinateFrame.lookVector:Dot(Vector3.new(0,1,0)))
			end

			function GetCharacterMass()
				return Services.Character.HumanoidRootPart:GetMass() + Services.Character.Head:GetMass()
			end

			function Magnitude2D(x,z)
				return math.sqrt(x*x+z*z)
			end

			local InputKeys = {
				["W"] = false,
				["S"] = false,
				["A"] = false,
				["D"] = false,
				["Space"] = false,
				["LMB"] = false,
				["RMB"] = false
			}

			local camera = workspace.CurrentCamera
			local UserInputService = game:GetService("UserInputService")
			function onInput(input, gameProcessedEvent)
				local InputState
				--print(input.KeyCode)
				if input.UserInputState == Enum.UserInputState.Begin then
					InputState = true
				elseif input.UserInputState == Enum.UserInputState.End then
					InputState = false
				else
					return
				end 

				if input.UserInputType == Enum.UserInputType.Keyboard then
					local key = input.KeyCode.Name
					if InputKeys[key] ~= nil then
						InputKeys[key] = InputState
					end
				elseif input.UserInputType == Enum.UserInputType.MouseButton1 then --LMB down
					InputKeys.LMB = InputState
				elseif input.UserInputType == Enum.UserInputType.MouseButton2 then --RMB down
					InputKeys.RMB = InputState
				end
			end
			function main()
				local A = Services.Character:FindFirstChildOfClass("Humanoid") or Services.LocalPlayer.Character:WaitForChild("Humanoid");
				A.PlatformStand = true
				Init(Services.LocalPlayer, Services.Camera, InputKeys);
				InitBodyMovers();

				UserInputService.InputBegan:connect(onInput);
				UserInputService.InputEnded:connect(onInput);
				--connect updateloop
				Services.RunService:BindToRenderStep("updateLoop", 1, updateLoop);

				--rip
			end

			local prevUpdateTime = nil
			local updateDT = 1/60

			function setDeltaTime()
				local UpdateTime = tick() 
				if prevUpdateTime ~= nil then
					updateDT = (UpdateTime - prevUpdateTime)
				else
					updateDT = 1/60
				end
				prevUpdateTime = UpdateTime
			end
			function updateLoop()
				setDeltaTime();
				update(updateDT);
			end
			main()
		end)
	end
)

				else
					make_writeable(table)
				end
			end)
			local GetRawMetatable = (debug and debug.getmetatable) or getrawmetatable
			local GameMetaTable = GetRawMetatable(game)
			local OIndex = GameMetaTable.__index

			SetReadOnly(GameMetaTable, false)

			GameMetaTable.__index =
				newcclosure(
					function(S, K)
						if string.lower(tostring(K)) == string.lower(tostring(Property)) then
						K = Value
					end
						return OIndex(S, K)
					end
				)
			SetReadOnly(GameMetaTable, true)
		else
			UIFuncs.SendNotification("Mada Admin", "Missing an argument", 5)
		end
	end
)

		Gravity = tonumber(Gravity) or 0

			if not Part or not Part:IsA("BasePart") or Part.Anchored then

			end

			if Services.Character and Part:IsDescendantOf(Services.Character) then
				return
			end

			if not Force then
				Force = UIFuncs.CreateInstance("BodyForce", {Parent = Part, Name = "MadaGravityForce"})

			function(Part)
				SetGrav(Part)
			end
		)

		for i, Descendant in next, workspace:GetDescendants() do
			SetGrav(Descendant)
		end

		Runtime.RegisterCleanup(
			"UnanchoredGravity",
			function()
				for _, Descendant in next, workspace:GetDescendants() do
					local Force = Descendant:FindFirstChild("MadaGravityForce")
					if Force then
						Force:Destroy()

							if Object and Object.Parent then
								Object:Destroy()

						end
					end

				CharacterTorso.Anchored = true

				AP.MaxForce = math.huge
				AO.MaxTorque = math.huge
				AO.Responsiveness = 200
				AP.Attachment0 = A0
				AP.Attachment1 = A1
				AO.Attachment1 = A1
				AO.Attachment0 = A0

				Runtime.StartLoop(
					"ControlNPCsNoCollide",
					function(Token)
						while Token.Active and CommandToggles.ControlNPCs and NPC and NPC.Parent do
							for _, Descendant in next, NPC:GetDescendants() do
								if Descendant:IsA("BasePart") then
									Descendant.CanCollide = false
								end
							end
							Services.RunService.Stepped:Wait()
						end
					end
				)
			end
		)

		UIFuncs.SendNotification(
			"Mada Admin",
			"ControlNPCs enabled, hold ctrl and click an NPC to control it",
			5
		)

Command.Add(
	{"ServerHop", "Shop", "Hop", "SH"},
	{"ServerHop / Shop / Hop / SH", "Teleports you into a different server for the current game"},
	{"None"},
	function()
		local Success, Response =
			pcall(
				function()
					return game:HttpGet(
						"https://games.roblox.com/v1/games/" .. tostring(game.PlaceId) .. "/servers/Public?sortOrder=Asc&limit=100"
					)
				end
			)

		if not Success or not Response then
			UIFuncs.SendNotification("Mada Admin", "Could not fetch public servers", 5)
			return
		end

		local DecodedSuccess, Data =
			pcall(function()
				return Services.HttpService:JSONDecode(Response)
			end)

		if not DecodedSuccess or type(Data) ~= "table" or type(Data.data) ~= "table" then
			UIFuncs.SendNotification("Mada Admin", "Failed to decode server list", 5)
			return
		end

		local Candidates = {}
		for _, Server in next, Data.data do
			if
				type(Server) == "table" and Server.id and Server.id ~= game.JobId and tonumber(Server.playing) and
					tonumber(Server.maxPlayers) and tonumber(Server.playing) < tonumber(Server.maxPlayers)
			then
				table.insert(Candidates, Server)
			end
		end

		if #Candidates == 0 then
			UIFuncs.SendNotification("Mada Admin", "No other server was found right now", 5)
			return
		end

		local ChosenServer = Candidates[math.random(1, #Candidates)]
		UIFuncs.SendNotification("Mada Admin", "Joining another server...", 4)
		Services.TeleportService:TeleportToPlaceInstance(game.PlaceId, ChosenServer.id, Services.LocalPlayer)
	end
)

	{"Username/All"},
	function(Target)
		if not Target or Target == "" then
			for i, Instance in next, Services.Character:FindFirstChild("HumanoidRootPart"):GetChildren() do
				if Instance.Name == "Spin" then
					Instance:Destroy()
				end
			end

			local Spin = Instance.new("BodyAngularVelocity")
			Spin.MaxTorque = Vector3.new(0, 9e9, 0)
			Spin.AngularVelocity = Vector3.new(0, 35, 0)
			Spin.Name = "Spin"
			Spin.Parent = Services.Character.HumanoidRootPart
			UIFuncs.SendNotification("Mada Admin", "Fling without target spins <b>you</b>", 4)
			return
		end

		local Targets = PlayerFuncs.GetPlayer(Target)
		local FilteredTargets = {}
		for _, Player in next, Targets do
			if Player and Player ~= Services.LocalPlayer then
				table.insert(FilteredTargets, Player)
			end
		end

		if #FilteredTargets == 0 then
			UIFuncs.SendNotification("Mada Admin", "No valid target found for fling", 4)
			return
		end

		local Success = PlayerFuncs.Fling(FilteredTargets)
		if Success then
			if tostring(Target):lower() == "all" then
				UIFuncs.SendNotification("Mada Admin", "Attempted to fling <b>all</b>", 4)
			else
				UIFuncs.SendNotification("Mada Admin", "Attempted to fling <b>" .. tostring(Target) .. "</b>", 4)
			end
		else
			UIFuncs.SendNotification("Mada Admin", "Fling failed on the selected target(s)", 4)
		end
	end
)

GUI = {}

GUI.Top = function(Method)
	local TopFrame = Interface and Interface:FindFirstChild("Top")
	if not TopFrame then
		return
	end

	if Method == "Open" or Method == "open" then
		UIFuncs.Open(TopFrame, 287, 39, UIFuncs.FrameSpeed)
	else
		UIFuncs.Close(TopFrame, UIFuncs.CloseSpeed)
	end
end

GUI.Commands = function()
	if not Tabs.Commands.Visible then
		Tabs.Commands.Visible = true
	end
	for i, v in next, Tabs.Commands:GetChildren() do
		if v:IsA("TextLabel") then
			GUI.Destroy(v)
		end
	end
	local i = 0
	for cmdName, tbl in next, Commands do
		UIFuncs.AddCommand(tbl[2][1], tbl[2][2], "Arguments: " .. tbl[3][1])
	end
end

GUI.Commands()

GUI.Tween = function(Object, Style, Direction, Duration, Goal)
	local TweenInfo = TweenInfo.new(Duration, Enum.EasingStyle[Style], Enum.EasingDirection[Direction])
	local Tween = Services.TweenService:Create(Object, TweenInfo, Goal)
	Tween:Play()
	return Tween
end

GUI.LoadCommands = function()
	for i, v in next, CommandAutoFill:GetChildren() do
		if v.Name ~= "UIListLayout" then
			UIFuncs.Destroy(v)
		end
	end
	local last = nil
	local i = 0
	for Name, Table in next, Commands do
		local BTN = CommandExample:Clone()
		BTN.Parent = CommandAutoFill
		BTN.Name = Name
		BTN.Input.Text = Table[2][1]
		i = i + 1

		local size = BTN.Size
		BTN.Size = UDim2.new(0, 0, 0, 25)
		BTN.Size = size
	end
end

GUI.SearchCommands = function()
	local str = (CommandInput.Text:gsub(";", "")):lower()
	local index = 0
	local lastFrame
	for _, v in next, CommandAutoFill:GetChildren() do
		if v:IsA("Frame") and index < 5 then
			local cmd = Commands[v.Name]
			--local name = cmd and cmd[2][1] or ""
			--v.Input.Text = str ~= "" and v.Name:find(str) == 1 and v.Name or name
			v.Visible = str == "" or v.Name:find(str)
			if v.Visible then
				index = index + 1
				local n = math.sqrt(index) * 125
				local yPos = (index - 1) * 28
				local newPos = UDim2.new(0.5, 0, 0, yPos)
				UIFuncs.Tween(
					v,
					UIFuncs.SmoothStyles.AutoFillOpen,
					{
						Size = UDim2.new(0.5, n, 0, 25),
						Position = lastFrame and newPos or UDim2.new(0.5, 0, 0, yPos)
					}
				)
				lastFrame = v
			end
		end
	end
end

GUI.LoadCommands()

for _, AutoFill in next, CommandAutoFill:GetChildren() do
	if AutoFill:IsA("Frame") then
		AutoFill.Visible = false
	end
end

GUI.BarSelect = function(Speed)
	local OpenDuration = Speed ~= nil and Speed or math.max(0.22, UIFuncs.FrameSpeed * 0.6)
	local OpenTween =
		Speed ~= nil and TweenInfo.new(OpenDuration, Enum.EasingStyle.Exponential, Enum.EasingDirection.Out) or
		UIFuncs.SmoothStyles.CmdBarOpen
	LeftFill.Visible = true
	RightFill.Visible = true
	CenterBar.Visible = true
	RefreshCommandInputVisibility()
	UIFuncs.Tween(CenterBar, OpenTween, {Size = UDim2.new(0, 250, 1, 15)})
	UIFuncs.Tween(LeftFill, OpenTween, {Position = UDim2.new(0, 0, 0.5, 0)})
	UIFuncs.Tween(RightFill, OpenTween, {Position = UDim2.new(1, 0, 0.5, 0)})
end

GUI.BarDeselect = function(Speed)
	local CloseDuration = Speed ~= nil and Speed or math.max(0.16, UIFuncs.CloseSpeed * 0.78)
	local CloseTween =
		Speed ~= nil and TweenInfo.new(CloseDuration, Enum.EasingStyle.Quint, Enum.EasingDirection.In) or
		UIFuncs.SmoothStyles.CmdBarClose
	UIFuncs.Tween(CenterBar, CloseTween, {Size = UDim2.new(0, 250, 0, 0)})
	UIFuncs.Tween(LeftFill, CloseTween, {Position = UDim2.new(-0.5, 100, 0.5, 0)})
	UIFuncs.Tween(RightFill, CloseTween, {Position = UDim2.new(1.5, -100, 0.5, 0)})
	task.spawn(function()
		task.wait(CloseDuration + 0.05)
		CenterBar.Visible = false
		LeftFill.Visible = false
		RightFill.Visible = false
	end)
	local AutoFillIndex = 0
	for i, v in next, CommandAutoFill:GetChildren() do
		if v:IsA("Frame") then
			AutoFillIndex = AutoFillIndex + 1
			task.spawn(
				function()
					task.wait((AutoFillIndex - 1) * 0.012)
					UIFuncs.Tween(v, UIFuncs.SmoothStyles.AutoFillClose, {Size = UDim2.new(0, 0, 0, 25)})
				end
			)
		end
	end
end

Services.Mouse.KeyDown:Connect(
	function(k)
		if k:lower() == Settings.Prefix then
			GUI.Top("Open")
			GUI.BarSelect()
			CommandInput.Text = ""
			CommandInput:CaptureFocus()
			task.wait()
			CommandInput.Text = ""
		end
	end
)

CommandInput.FocusLost:Connect(
	function(EnterPressed)
		if EnterPressed then
			spawn(
				function()
					Library.ParseCommand(Settings.Prefix .. CommandInput.Text)
				end
			)
		end
		GUI.Top("Close")
		GUI.BarDeselect()
	end
)

CommandInput.Changed:Connect(
	function(Changed)
		if Changed ~= "Text" then
			return
		end
		GUI.SearchCommands()
	end
)
